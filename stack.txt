//write a program to demonstrate a stack.
#include<iostream.h>
#include<conio.h>
int stack[100],n=100,top=-1;
void push(int val){
if(top>=n-1){
cout<<"Stack overflow"<<endl;
}
else{
top++;
stack[top]=val;
}
}
void pop(){
if(top<=-1){
cout<<"Stack underflow"<<endl;
}
else{
cout<<"The popped element is:-"<<stack[top]<<endl;
}
}
void display(){
if(top>=0){
cout<<"stack elements are:-";
for(int i=top;i>=0;i--){
cout<<stack[i]<<" ";
}
cout<<endl;
}
else{
cout<<"stack is empty";
}
}
